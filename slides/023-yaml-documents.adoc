    tls.crt: mon-certificat-preprod
    tls.key: ma-cle-preprod

    name: preprod-tls
type: kubernetes.io/tls
---

    tls.crt: mon-certificat-prod
    tls.key: ma-cle-prod

    name: prod-tls

----
